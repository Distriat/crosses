// –ü—Ä–æ–¥–≤–∏–Ω—É—Ç—ã–π —Ä–∞—Å—á–µ—Ç –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ Buck converter
class BuckConverterDesign {
  constructor(Vin, Vout, Iout, Fsw) {
    this.Vin = Vin;
    this.Vout = Vout;
    this.Iout = Iout;
    this.Fsw = Fsw;
  }

  calculateInductor() {
    const D = this.Vout / this.Vin;
    const L = (this.Vin - this.Vout) * D / (this.Fsw * 0.3 * this.Iout);
    return Math.max(L, 0.1); // ¬µH
  }

  calculateOutputCap(ripple) {
    const D = this.Vout / this.Vin;
    const Cout = (1 - D) / (8 * this.Fsw * ripple * this.Vout);
    return Cout * 1e6; // ¬µF
  }
}
